186static inline cpVect
187cpClosetPointOnSegment(const cpVect p, const cpVect a, const cpVect b)
188{
189 cpVect delta = cpvsub(a, b);
190 cpFloat t = cpfclamp01(cpvdot(delta, cpvsub(p, b))/cpvlengthsq(delta));
191 return cpvadd(b, cpvmult(delta, t));
192}
