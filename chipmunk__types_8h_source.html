119static inline cpFloat cpfmax(cpFloat a, cpFloat b)
120{
121 return (a > b) ? a : b;
122}
123
125static inline cpFloat cpfmin(cpFloat a, cpFloat b)
126{
127 return (a < b) ? a : b;
128}
129
131static inline cpFloat cpfabs(cpFloat f)
132{
133 return (f < 0) ? -f : f;
134}
135
137static inline cpFloat cpfclamp(cpFloat f, cpFloat min, cpFloat max)
138{
139 return cpfmin(cpfmax(f, min), max);
140}
141
143static inline cpFloat cpfclamp01(cpFloat f)
144{
145 return cpfmax(0.0f, cpfmin(f, 1.0f));
146}

149
151static inline cpFloat cpflerp(cpFloat f1, cpFloat f2, cpFloat t)
152{
153 return f1*(1.0f - t) + f2*t;
154}
155
157static inline cpFloat cpflerpconst(cpFloat f1, cpFloat f2, cpFloat d)
158{
159 return f1 + cpfclamp(f2 - f1, -d, d);
160}
